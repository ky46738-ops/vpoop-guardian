# 🛡 VPoopGuardian v1.0 - 開源版

import hashlib, json
from datetime import datetime
from pathlib import Path

VPOOP_CONSTITUTION = "人生自古誰無💩，不要把別人拉的說成自己拉的。"

class VPoopGuardian:
    STATES = ["IDLE", "DECLARED", "EXECUTING", "POOPED", "COMPLETED"]

    def __init__(self, target_files):
        self.targets = [Path(f) for f in target_files]
        self.state = "IDLE"
        self.baseline = {}

    def set_baseline(self, baseline_dict):
        self.baseline = baseline_dict

    def sha256sum(self, file_path):
        if not file_path.is_file(): return None
        h = hashlib.sha256()
        with open(file_path, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""): h.update(chunk)
        return h.hexdigest()

    def check_integrity(self):
        self.state = "EXECUTING"
        anomalies = []
        results = []
        for file_path in self.targets:
            current_hash = self.sha256sum(file_path)
            expected_hash = self.baseline.get(str(file_path))
            is_liar = expected_hash and current_hash != expected_hash
            if is_liar: anomalies.append(str(file_path))
            results.append({"file": str(file_path), "is_liar": is_liar})
        self.state = "POOPED" if anomalies else "COMPLETED"
        return {
            "guardian": "VPoopGuardian v1.0",
            "constitution": VPOOP_CONSTITUTION,
            "state": self.state,
            "liars_detected": anomalies,
            "details": results,
        }

if __name__ == "__main__":
    print("💩 VPoopGuardian v1.0 已啟動")
    guardian = VPoopGuardian(["我的合約.pdf"])
    guardian.set_baseline({"我的合約.pdf": "填入真實SHA256..."})
    report = guardian.check_integrity()
    print(json.dumps(report, ensure_ascii=False, indent=2))
